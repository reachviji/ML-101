# cdf gives you cummulative probability associated with a function
# Cumulative sum of area under curve upto gives you cdf
one = haberman.loc[haberman['survival_status']== 1]
two = haberman.loc[haberman['survival_status']== 2]
count,bins = np.histogram(one['axil_nodes'], density=True)
pdf = count/sum(count)
cdf = np.cumsum(pdf)
bins
# Plot the graph
plt.plot(bins[1:],cdf)
plt.xlabel('Age of survived')
